<template>
  <div class="background-image">
    <div id="app">
      <div class="frame">
        <h1 class="app-title">GFF3 Formatter</h1>
        <div class="bkgnd">
          <input type="file" @change="onFileChange" />
          <button @click="uploadFile" class="custom-button">Upload and Process</button>
          <div class="downloadlink">
            <a v-if="downloadLink" :href="downloadLink" :download="formatted_output">
              Download {{ formatted_output }}
            </a>
          </div>
        </div>
      </div>
    </div>
    <footer class="footer">
      <div class="container">
        <span class="text-muted">Your Footer Content Here</span>
      </div>
    </footer>
  </div>
</template>

<script>
export default {
  data() {
    return {
      selectedFile: null,
      downloadLink: null,
      formatted_output: '' // Initialize with an empty string
    };
  },
  methods: {
    onFileChange(event) {
      this.selectedFile = event.target.files[0];
    },
    async uploadFile() {
      if (!this.selectedFile) {
        alert('Please select a file first.');
        return;
      }

      const formData = new FormData();
      formData.append('file', this.selectedFile);

      try {
        const response = await fetch('http://127.0.0.1:5000/upload', {
          method: 'POST',
          body: formData,
        });
        const result = await response.json();
        if (result.formatted_output_path) {
          this.formatted_output = result.formatted_output_path; // Update formatted_output with the filename
          this.downloadLink = `http://127.0.0.1:5000/download/${result.formatted_output_path}`;
        } else {
          alert('Error processing file.');
        }
      } catch (error) {
        console.error('Error uploading file:', error);
        alert('Error uploading file.');
      }
    },
  },
};
</script>
